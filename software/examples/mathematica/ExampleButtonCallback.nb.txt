Needs["NETLink`"]
LoadNETAssembly["Tinkerforge",NotebookDirectory[]<>"../../.."]

host="localhost"
port=4223
uid="SCD32"(* Change to your UID *)

(* Create IPConnection and device object *)
ipcon=NETNew["Tinkerforge.IPConnection"]
lcd=NETNew["Tinkerforge.BrickletLCD16x4",uid,ipcon]
ipcon@Connect[host,port]

(* Callback functions for button status *)
ButtonPressedCB[sender_,button_]:=Print["Pressed: "<>ToString[button]]
AddEventHandler[lcd@ButtonPressed,ButtonPressedCB]

ButtonReleasedCB[sender_,button_]:=Print["Released: "<>ToString[button]]
AddEventHandler[lcd@ButtonReleased,ButtonReleasedCB]

Input["Click OK to exit"]

(* Clean up *)
ipcon@Disconnect[]
ReleaseNETObject[lcd]
ReleaseNETObject[ipcon]
